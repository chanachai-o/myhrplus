Diagrams Component Demo

A comprehensive diagram component for creating flowcharts, organizational charts, network diagrams, and more.

Basic Usage 

// TypeScript
import {{ '{' }} DiagramsComponent {{ '}' }} from '../../shared/components/diagrams/diagrams.component';
import {{ '{' }} NodeModel, ConnectorModel {{ '}' }} from '@syncfusion/ej2-angular-diagrams';

@ViewChild('diagram') diagram!: DiagramsComponent;

nodes: NodeModel[] = [
  {{ '{' }}
    id: 'node1',
    offsetX: 100,
    offsetY: 100,
    width: 100,
    height: 50,
    shape: {{ '{' }} type: 'Basic', shape: 'Rectangle' {{ '}' }},
    annotations: [{{ '{' }} content: 'Node 1' {{ '}' }}]
  {{ '}' }}
];

connectors: ConnectorModel[] = [
  {{ '{' }}
    id: 'connector1',
    sourceID: 'node1',
    targetID: 'node2',
    type: 'Orthogonal'
  {{ '}' }}
];

// HTML
<app-diagrams
  #diagram
  [nodes]="nodes"
  [connectors]="connectors"
  [height]="'600px'"
  [width]="'100%'">
</app-diagrams>

  • Export (Image, SVG)
  • Print support
  • Grid and ruler settings
  • Layout algorithms
  • Data binding support
