Image Editor Component Demo

Component สำหรับแก้ไขรูปภาพจาก Syncfusion พร้อม Glass Morphism styling และ features ครบถ้วน

ตัวอย่างการใช้งาน

// TypeScript
import {{ '{' }} ImageEditorComponent {{ '}' }} from '../../shared/components/image-editor/image-editor.component';

@ViewChild('editor') editor!: ImageEditorComponent;

// Open image
openImage(url: string): void {{ '{' }}
  this.editor.open(url);
{{ '}' }}

// Export
exportAsPng(): void {{ '{' }}
  this.editor.export('PNG', 'image');
{{ '}' }}

// Transform
rotateLeft(): void {{ '{' }}
  this.editor.transform('RotateLeft');
{{ '}' }}

// HTML
<app-image-editor
  #editor
  [width]="'100%'"
  [height]="'600px'"
  [src]="imageUrl"
  [toolbar]="toolbarItems"
  [allowUndoRedo]="true"
  (created)="onCreated($event)"
  (imageLoaded)="onImageLoaded($event)">
</app-image-editor>
