Document Editor Component Demo

Component สำหรับแก้ไขเอกสาร Word-like จาก Syncfusion พร้อม Glass Morphism styling และ features ครบถ้วน

ตัวอย่างการใช้งาน

// TypeScript
import {{ '{' }} DocumentEditorComponent {{ '}' }} from '../../shared/components/document-editor/document-editor.component';

@ViewChild('editor') editor!: DocumentEditorComponent;

// Open document
openDocument(sfdt: string): void {{ '{' }}
  this.editor.open(sfdt);
{{ '}' }}

// Export
exportAsWord(): void {{ '{' }}
  this.editor.export('Docx', 'document');
{{ '}' }}

exportAsPdf(): void {{ '{' }}
  this.editor.export('Pdf', 'document');
{{ '}' }}

// Save as SFDT
saveDocument(): string {{ '{' }}
  return this.editor.saveAsSfdt();
{{ '}' }}

// Print
printDocument(): void {{ '{' }}
  this.editor.print();
{{ '}' }}

// HTML
<app-document-editor
  #editor
  [width]="'100%'"
  [height]="'600px'"
  [enableSpellCheck]="true"
  [enableEditor]="true"
  [showRuler]="true"
  (created)="onCreated($event)"
  (documentChange)="onDocumentChange($event)">
</app-document-editor>
